## 插入代码块
在第一行后写上对应语言能实现代码高亮
``` c++ 
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,m;
    cin>>n;
    m=n;
    bool flag=true;
    cout<<n<<"=";
    if(n==1) {cout<<1<<endl;return 0;}
    for(int i=2;i<=m/i;i++){
        int k=0;
        if(n%i==0){
            while(n%i==0){
                n=n/i;
                k++;
            }
            if(flag){cout<<i;flag=false;}
            else cout<<"*"<<i;
            if(k>1) cout<<"^"<<k;
        }
    }
    //如果最终的n大于1说明m也是其中一个质因子
    if(n>1){
        if(flag){cout<<n;flag=false;}
        else cout<<"*"<<n;
    }
    cout<<endl;
    return 0;
}
```
